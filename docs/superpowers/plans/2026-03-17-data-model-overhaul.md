```typescript
  pgTable, uuid, varchar, text, date, integer, decimal,

import { relations } from "drizzle-orm";

// Enums
export const tripRoleEnum = pgEnum("trip_role", ["owner", "viewer"]);

export const users = pgTable("users", {
  id: uuid("id").primaryKey().defaultRandom(),
  email: varchar("email", { length: 255 }).notNull().unique(),
  name: varchar("name", { length: 255 }),
  avatarUrl: varchar("avatar_url", { length: 500 }),
  preferences: jsonb("preferences").$type<{ theme?: string }>(),
  createdAt: timestamp("created_at", { withTimezone: true }).notNull().defaultNow(),
  updatedAt: timestamp("updated_at", { withTimezone: true }).notNull().defaultNow(),
});

export const trips = pgTable("trips", {
  id: uuid("id").primaryKey().defaultRandom(),
  name: varchar("name", { length: 255 }).notNull(),
  description: text("description"),
  startDate: date("start_date"),
  endDate: date("end_date"),
  defaultCurrency: varchar("default_currency", { length: 3 }).notNull().default("USD"),
  coverImageUrl: varchar("cover_image_url", { length: 500 }),
  isPublic: boolean("is_public").notNull().default(false),
  createdAt: timestamp("created_at", { withTimezone: true }).notNull().defaultNow(),
  updatedAt: timestamp("updated_at", { withTimezone: true }).notNull().defaultNow(),
});

export const tripMembers = pgTable("trip_members", {
  tripId: uuid("trip_id").notNull().references(() => trips.id, { onDelete: "cascade" }),
  userId: uuid("user_id").notNull().references(() => users.id, { onDelete: "cascade" }),
  role: tripRoleEnum("role").notNull(),

]);

export const days = pgTable("days", {
  id: uuid("id").primaryKey().defaultRandom(),
  tripId: uuid("trip_id").notNull().references(() => trips.id, { onDelete: "cascade" }),
  date: date("date"),
  label: varchar("label", { length: 255 }),
  sortOrder: integer("sort_order").notNull().default(0),
  createdAt: timestamp("created_at", { withTimezone: true }).notNull().defaultNow(),
  updatedAt: timestamp("updated_at", { withTimezone: true }).notNull().defaultNow(),
}, (t) => [
  index("days_trip_sort_idx").on(t.tripId, t.sortOrder),
]);

export const categories = pgTable("categories", {
  id: uuid("id").primaryKey().defaultRandom(),
  tripId: uuid("trip_id").references(() => trips.id, { onDelete: "cascade" }),

  label: varchar("label", { length: 100 }).notNull(),
  color: varchar("color", { length: 7 }).notNull(),
  updatedAt: timestamp("updated_at", { withTimezone: true }).notNull().defaultNow(),
}, (t) => [
  uniqueIndex("categories_trip_slug_idx").on(t.tripId, t.slug),
]);

export const items = pgTable("items", {
  id: uuid("id").primaryKey().defaultRandom(),
  tripId: uuid("trip_id").notNull().references(() => trips.id, { onDelete: "cascade" }),

  categoryId: uuid("category_id").notNull().references(() => categories.id),
  name: varchar("name", { length: 255 }).notNull(),
  sortOrder: integer("sort_order").notNull().default(0),
  address: varchar("address", { length: 500 }),
  lat: decimal("lat", { precision: 10, scale: 7 }),
  lng: decimal("lng", { precision: 10, scale: 7 }),
  placeId: varchar("place_id", { length: 255 }),
  startTime: timestamp("start_time", { withTimezone: true }),
  endTime: timestamp("end_time", { withTimezone: true }),
  costAmount: integer("cost_amount"),
  costCurrency: varchar("cost_currency", { length: 3 }),
  notes: text("notes"),
  createdBy: uuid("created_by").notNull().references(() => users.id),
  assignedTo: uuid("assigned_to").references(() => users.id),
  createdAt: timestamp("created_at", { withTimezone: true }).notNull().defaultNow(),
  updatedAt: timestamp("updated_at", { withTimezone: true }).notNull().defaultNow(),
}, (t) => [
  index("items_trip_day_idx").on(t.tripId, t.dayId),
  index("items_trip_category_idx").on(t.tripId, t.categoryId),
]);

// Relations
export const usersRelations = relations(users, ({ many }) => ({
  tripMembers: many(tripMembers),
  createdItems: many(items, { relationName: "createdBy" }),
  assignedItems: many(items, { relationName: "assignedTo" }),
}));

export const tripsRelations = relations(trips, ({ many }) => ({
  members: many(tripMembers),
  days: many(days),
  items: many(items),
  categories: many(categories),
}));

export const tripMembersRelations = relations(tripMembers, ({ one }) => ({
  trip: one(trips, { fields: [tripMembers.tripId], references: [trips.id] }),
  user: one(users, { fields: [tripMembers.userId], references: [users.id] }),
}));

export const daysRelations = relations(days, ({ one, many }) => ({
  trip: one(trips, { fields: [days.tripId], references: [trips.id] }),
  items: many(items),
}));

export const categoriesRelations = relations(categories, ({ one, many }) => ({
  trip: one(trips, { fields: [categories.tripId], references: [trips.id] }),
  items: many(items),
}));

export const itemsRelations = relations(items, ({ one }) => ({
  trip: one(trips, { fields: [items.tripId], references: [trips.id] }),
  day: one(days, { fields: [items.dayId], references: [days.id] }),
  category: one(categories, { fields: [items.categoryId], references: [categories.id] }),
  creator: one(users, { fields: [items.createdBy], references: [users.id], relationName: "createdBy" }),
  assignee: one(users, { fields: [items.assignedTo], references: [users.id], relationName: "assignedTo" }),
```

```typescript
const CURRENCY_SYMBOLS: Record<string, string> = {
  "$": "USD", "€": "EUR", "£": "GBP", "¥": "JPY", "₩": "KRW",
  "A$": "AUD", "C$": "CAD", "₹": "INR", "R$": "BRL", "₱": "PHP",
};

/**
 * Parse a cost string like "$50", "€12.50", "¥3000" into cents + currency.
 * Returns null if unparseable.
 */
export function parseCostString(
  raw: string,
  fallbackCurrency = "USD",
): { amount: number; currency: string } | null {
  const trimmed = raw.trim();
  if (!trimmed) return null;

  // Try to match a currency symbol at the start
  let currency = fallbackCurrency;
  let numStr = trimmed;

  for (const [symbol, code] of Object.entries(CURRENCY_SYMBOLS)) {
    if (trimmed.startsWith(symbol)) {
      currency = code;
      numStr = trimmed.slice(symbol.length).trim();
      break;
    }
  }

  const num = parseFloat(numStr.replace(/,/g, ""));
  if (isNaN(num)) return null;

  // JPY and KRW don't use subunits
  const noSubunit = ["JPY", "KRW", "VND"];
  const amount = noSubunit.includes(currency) ? Math.round(num) : Math.round(num * 100);

  return { amount, currency };
}

/**
 * Format cents + currency back to display string.
 */
export function formatCost(amount: number, currency: string): string {
  const noSubunit = ["JPY", "KRW", "VND"];
  const value = noSubunit.includes(currency) ? amount : amount / 100;
  try {
    return new Intl.NumberFormat("en-US", { style: "currency", currency }).format(value);
  } catch {
    return `${value} ${currency}`;
  }
}
```

```typescript
import { parse, isValid, format } from "date-fns";

const DATE_FORMATS = [
  "MMM d yyyy h:mma",
  "MMM d yyyy h:mm a",
  "MMM d yyyy",
  "MMM d h:mma",
  "MMM d h:mm a",
  "MMM d",
  "yyyy-MM-dd'T'HH:mm:ss",
  "yyyy-MM-dd",
];

/**
 * Best-effort parse a free-text date/time string.
 * Returns an ISO timestamp string or null.
 */
export function parseDateString(raw: string): string | null {
  const trimmed = raw.trim();
  if (!trimmed) return null;

  // Try ISO first
  const isoDate = new Date(trimmed);
  if (isValid(isoDate) && trimmed.includes("-")) {
    return isoDate.toISOString();
  }

  // Try each format
  const now = new Date();
  const currentYear = now.getFullYear();

  for (const fmt of DATE_FORMATS) {
    const parsed = parse(trimmed, fmt, now);
    if (isValid(parsed)) {
      // If no year in format, assume current year
      if (!fmt.includes("yyyy") && parsed.getFullYear() === 2001) {
        parsed.setFullYear(currentYear);
      }
      return parsed.toISOString();
    }
  }

  return null;
}

/**
 * Format a timestamp for display.
 */
export function formatDateTime(date: Date | string): string {
  const d = typeof date === "string" ? new Date(date) : date;
  if (!isValid(d)) return "";
  return format(d, "MMM d, yyyy h:mm a");
}

export function formatDateOnly(date: Date | string): string {
  const d = typeof date === "string" ? new Date(date) : date;
  if (!isValid(d)) return "";
  return format(d, "MMM d, yyyy");
}
```

**Auth schema reconciliation:** The Auth.js Drizzle adapter requires specific tables (`users`, `accounts`, `sessions`, `verification_tokens`). Our approach:

1. Use the adapter's expected schema as the base for `users` — it provides `id`, `name`, `email`, `emailVerified`, `image`.
2. Add our custom columns (`preferences` JSONB, `avatarUrl` aliased from `image`) via a Drizzle migration after initial setup.
3. Define `accounts`, `sessions`, and `verification_tokens` tables in `schema.ts` following the [Auth.js Drizzle adapter docs](https://authjs.dev/getting-started/adapters/drizzle).
4. Map `avatarUrl` to the adapter's `image` field (they serve the same purpose).
5. The `preferences` column is added as nullable JSONB — existing adapter functionality is unaffected.

In practice, this means the `users` table in Task 2 should include the adapter-required columns (`emailVerified TIMESTAMPTZ`, `image VARCHAR`) alongside our custom `preferences` column. The adapter will work with this extended schema.

```typescript
"use client";

import { useState, useCallback, useRef } from "react";

export type OutputLine = {
  id: number;
  text: string;
  className?: string;
  html?: boolean;
};

type PromptState = {
  message: string;
  resolve: (value: string) => void;
} | null;

export function useTerminal() {
  const [output, setOutput] = useState<OutputLine[]>([]);
  const [history, setHistory] = useState<string[]>([]);
  const [historyIndex, setHistoryIndex] = useState(-1);
  const [isProcessing, setIsProcessing] = useState(false);
  const [promptState, setPromptState] = useState<PromptState>(null);
  const nextId = useRef(0);

  const print = useCallback((text: string, className?: string) => {
    setOutput((prev) => [...prev, { id: nextId.current++, text, className }]);
  }, []);

  const printHtml = useCallback((html: string, className?: string) => {
    setOutput((prev) => [...prev, { id: nextId.current++, text: html, className, html: true }]);
  }, []);

  const clear = useCallback(() => {
    setOutput([]);
  }, []);

  const prompt = useCallback((message: string): Promise<string> => {
    return new Promise((resolve) => {
      setPromptState({ message, resolve });
    });
  }, []);

  const resolvePrompt = useCallback((value: string) => {
    if (promptState) {
      promptState.resolve(value);
      setPromptState(null);
    }
  }, [promptState]);

  const addToHistory = useCallback((cmd: string) => {
    setHistory((prev) => {
      const filtered = prev.filter((h) => h !== cmd);
      return [...filtered, cmd];
    });
    setHistoryIndex(-1);
  }, []);

  const navigateHistory = useCallback((direction: "up" | "down"): string | null => {
    if (history.length === 0) return null;

    let newIndex: number;
    if (direction === "up") {
      newIndex = historyIndex === -1 ? history.length - 1 : Math.max(0, historyIndex - 1);
    } else {
      newIndex = historyIndex === -1 ? -1 : historyIndex + 1;
      if (newIndex >= history.length) newIndex = -1;
    }

    setHistoryIndex(newIndex);
    return newIndex === -1 ? "" : history[newIndex];
  }, [history, historyIndex]);

  return {
    output,
    isProcessing,
    setIsProcessing,
    promptState,
    print,
    printHtml,
    clear,
    prompt,
    resolvePrompt,
    addToHistory,
    navigateHistory,
  };
}
```

```tsx
"use client";

import { useRef, useEffect, useState, type KeyboardEvent } from "react";

type Props = {
  onSubmit: (input: string) => void;
  onHistoryNav: (direction: "up" | "down") => string | null;
  disabled: boolean;
  promptMessage?: string;
};

export function TerminalInput({ onSubmit, onHistoryNav, disabled, promptMessage }: Props) {
  const inputRef = useRef<HTMLInputElement>(null);
  const [value, setValue] = useState("");

  useEffect(() => {
    inputRef.current?.focus();
  }, [disabled]);

  const handleKeyDown = (e: KeyboardEvent<HTMLInputElement>) => {
    if (e.key === "Enter" && !disabled) {
      onSubmit(value);
      setValue("");
    } else if (e.key === "ArrowUp") {
      e.preventDefault();
      const cmd = onHistoryNav("up");
      if (cmd !== null) setValue(cmd);
    } else if (e.key === "ArrowDown") {
      e.preventDefault();
      const cmd = onHistoryNav("down");
      if (cmd !== null) setValue(cmd);
    }
  };

  const prefix = promptMessage || "guest@trip-terminal:~$";

  return (
    <div
      className="terminal-input-line"
      onClick={() => inputRef.current?.focus()}
    >
      <span className="terminal-prompt">{prefix}</span>
      <span className="terminal-input-wrapper">
        <input
          ref={inputRef}
          type="text"
          value={value}
          onChange={(e) => setValue(e.target.value)}
          onKeyDown={handleKeyDown}
          disabled={disabled}
          className="terminal-hidden-input"
          autoComplete="off"
          spellCheck={false}
        />
        <span className="terminal-mirror">{value}</span>
        <span className="terminal-cursor" />
      </span>
    </div>
  );
}
```

```tsx
"use client";

import { useRef, useEffect, useCallback } from "react";
import { useTerminal } from "./useTerminal";
import { TerminalInput } from "./TerminalInput";
import { TerminalOutput } from "./TerminalOutput";

type Props = {
  tripId: string;
  onRefresh: () => void;
};

export function Terminal({ tripId, onRefresh }: Props) {
  const terminal = useTerminal();
  const scrollRef = useRef<HTMLDivElement>(null);

  // Auto-scroll to bottom on new output
  useEffect(() => {
    if (scrollRef.current) {
      scrollRef.current.scrollTop = scrollRef.current.scrollHeight;
    }
  }, [terminal.output]);

  const handleSubmit = useCallback(async (input: string) => {
    const trimmed = input.trim();
    if (!trimmed) return;

    // If in prompt mode, resolve the prompt
    if (terminal.promptState) {
      terminal.resolvePrompt(trimmed);
      return;
    }

    terminal.addToHistory(trimmed);
    terminal.print(`$ ${trimmed}`, "dim");
    terminal.setIsProcessing(true);

    try {
      // Dynamic import to avoid circular deps
      const { executeCommand } = await import("@/lib/commands");
      await executeCommand(trimmed, tripId, {
        print: terminal.print,
        printHtml: terminal.printHtml,
        clear: terminal.clear,
        prompt: terminal.prompt,
        refresh: onRefresh,
      });
    } catch (err: any) {
      terminal.print(err.message || "Command failed", "error");
    } finally {
      terminal.setIsProcessing(false);
    }
  }, [terminal, tripId, onRefresh]);

  return (
    <div className="terminal" ref={scrollRef}>
      <TerminalOutput lines={terminal.output} />
      <TerminalInput
        onSubmit={handleSubmit}
        onHistoryNav={terminal.navigateHistory}
        disabled={terminal.isProcessing}
        promptMessage={terminal.promptState?.message}
      />
    </div>
  );
}
```

```typescript
/**
 * Tokenize a command string with quote support.
 * "hello world" → single token. Handles both single and double quotes.
 */
export function tokenize(input: string): string[] {
  const tokens: string[] = [];
  let current = "";
  let inQuote: string | null = null;

  for (const char of input) {
    if (inQuote) {
      if (char === inQuote) {
        inQuote = null;
      } else {
        current += char;
      }
    } else if (char === '"' || char === "'") {
      inQuote = char;
    } else if (char === " ") {
      if (current) {
        tokens.push(current);
        current = "";
      }
    } else {
      current += char;
    }
  }

  if (current) tokens.push(current);
  return tokens;
}

export function parseCommand(input: string): { command: string; args: string[] } {
  const tokens = tokenize(input);
  if (tokens.length === 0) return { command: "", args: [] };

  const command = tokens[0].toLowerCase();
  const args = tokens.slice(1);

  return { command, args };
}
```

```typescript
commands["edit"] = async (args, tripId, ctx) => {
  if (args.length < 3) {
    ctx.print("Usage: edit <name> <field> <value>", "warning");
    ctx.print("Fields: name, address, time, cost, notes, category", "dim");
    return;
  }

  // Parse: last two tokens might be field + value, or field is second-to-last
  // Strategy: try to find item by progressively shorter name prefixes
  const items = await api(`/api/trips/${tripId}/items`);
  let match: any = null;
  let fieldIdx = -1;

  for (let i = 1; i < args.length - 1; i++) {
    const namePart = args.slice(0, i).join(" ").toLowerCase();
    const candidate = items.find((it: any) => it.name.toLowerCase() === namePart)
      || items.find((it: any) => it.name.toLowerCase().includes(namePart));
    if (candidate) {
      match = candidate;
      fieldIdx = i;
    }
  }

  if (!match || fieldIdx === -1) {
    ctx.print(`No item matching "${args[0]}"`, "error");
    return;
  }

  const field = args[fieldIdx].toLowerCase();
  const value = args.slice(fieldIdx + 1).join(" ");

  const updateBody: Record<string, any> = {};
  if (field === "name") updateBody.name = value;
  else if (field === "address") updateBody.address = value;
  else if (field === "notes") updateBody.notes = value;
  else if (field === "category") updateBody.categorySlug = value;
  else if (field === "cost") {
    const parsed = parseCostString(value);
    if (parsed) {
      updateBody.costAmount = parsed.amount;
      updateBody.costCurrency = parsed.currency;
    } else {
      ctx.print(`Could not parse cost "${value}"`, "error");
      return;
    }
  } else if (field === "time") {
    updateBody.startTime = value; // API will parse
  } else {
    ctx.print(`Unknown field "${field}"`, "error");
    return;
  }

  const updated = await api(`/api/trips/${tripId}/items/${match.id}`, {
    method: "PATCH",
    body: JSON.stringify(updateBody),
  });

  ctx.print(`Updated "${updated.name}" — ${field} = ${value}`, "success");
  ctx.refresh();
};
```

```typescript
"use client";

import { useRef, useCallback, useEffect, useState } from "react";
import type { ItemWithCategory } from "@/types";

type MarkerMap = Map<string, google.maps.marker.AdvancedMarkerElement>;

export function useMap(items: ItemWithCategory[]) {
  const mapRef = useRef<google.maps.Map | null>(null);
  const markersRef = useRef<MarkerMap>(new Map());
  const mapContainerRef = useRef<HTMLDivElement>(null);
  const [mapReady, setMapReady] = useState(false);

  const initMap = useCallback(async (container: HTMLDivElement) => {
    const { Map } = await google.maps.importLibrary("maps") as google.maps.MapsLibrary;

    mapRef.current = new Map(container, {
      center: { lat: 35.6762, lng: 139.6503 },
      zoom: 3,
      mapId: "trip-terminal-map",
      disableDefaultUI: true,
      zoomControl: true,
    });

    setMapReady(true);
  }, []);

  // Sync markers whenever items change
  useEffect(() => {
    if (!mapRef.current || !mapReady) return;

    const map = mapRef.current;
    const currentMarkers = markersRef.current;
    const itemIds = new Set(items.map((i) => i.id));

    // Remove markers for deleted items
    for (const [id, marker] of currentMarkers) {
      if (!itemIds.has(id)) {
        marker.map = null;
        currentMarkers.delete(id);
      }
    }

    // Add/update markers for items with coordinates
    for (const item of items) {
      if (!item.lat || !item.lng) continue;

      if (currentMarkers.has(item.id)) continue; // Already exists

      const pin = new google.maps.marker.PinElement({
        background: item.category.color,
        borderColor: item.category.color,
        glyphColor: "#ffffff",
      });

      const marker = new google.maps.marker.AdvancedMarkerElement({
        map,
        position: { lat: Number(item.lat), lng: Number(item.lng) },
        title: item.name,
        content: pin.element,
      });

      currentMarkers.set(item.id, marker);
    }
  }, [items, mapReady]);

  const flyTo = useCallback((item: ItemWithCategory) => {
    if (!mapRef.current || !item.lat || !item.lng) return;
    mapRef.current.panTo({ lat: Number(item.lat), lng: Number(item.lng) });
    mapRef.current.setZoom(15);
  }, []);

  const fitAll = useCallback(() => {
    if (!mapRef.current) return;
    const bounds = new google.maps.LatLngBounds();
    let count = 0;
    for (const item of items) {
      if (item.lat && item.lng) {
        bounds.extend({ lat: Number(item.lat), lng: Number(item.lng) });
        count++;
      }
    }
    if (count > 0) mapRef.current.fitBounds(bounds, 40);
  }, [items]);

  const toggleMapType = useCallback(() => {
    if (!mapRef.current) return;
    const current = mapRef.current.getMapTypeId();
    mapRef.current.setMapTypeId(
      current === "roadmap" ? "hybrid" : "roadmap"
    );
  }, []);

  return { mapContainerRef, initMap, flyTo, fitAll, toggleMapType, mapReady };
}
```

```typescript
type Message = { role: "user" | "assistant"; content: string };
type Suggestion = { name: string; category: string; address?: string; notes?: string };

export function useChat(tripId: string, onItemAdded: () => void) {
  const [messages, setMessages] = useState<Message[]>([]);
  const [suggestions, setSuggestions] = useState<Suggestion[]>([]);
  const [isLoading, setIsLoading] = useState(false);

  const send = useCallback(async (userMessage: string) => {
    const userMsg: Message = { role: "user", content: userMessage };
    const newMessages = [...messages, userMsg];
    setMessages(newMessages);
    setIsLoading(true);

    try {
      // Get trip context
      const items = await fetch(`/api/trips/${tripId}/items`).then((r) => r.json());
      const trip = await fetch(`/api/trips/${tripId}`).then((r) => r.json());

      const res = await fetch("/api/chat", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          messages: newMessages.map((m) => ({ role: m.role, content: m.content })),
          tripContext: { name: trip.name, items },
        }),
      });

      const data = await res.json();
      let assistantText = "";

      for (const block of data.content || []) {
        if (block.type === "text") {
          assistantText += block.text;
        } else if (block.type === "tool_use") {
          if (block.name === "add_item") {
            const input = block.input;
            await fetch(`/api/trips/${tripId}/items`, {
              method: "POST",
              headers: { "Content-Type": "application/json" },
              body: JSON.stringify({
                name: input.name,
                categorySlug: input.category,
                address: input.address,
                startTime: input.time ? parseDateString(input.time) : undefined,
                costAmount: input.cost ? parseCostString(input.cost)?.amount : undefined,
                costCurrency: input.cost ? parseCostString(input.cost)?.currency : undefined,
                notes: input.notes,
              }),
            });
            assistantText += `\n✅ Added "${input.name}" to your trip.`;
            onItemAdded();
          } else if (block.name === "suggest_items") {
            setSuggestions(block.input.items);
          }
        }
      }

      if (assistantText) {
        setMessages((prev) => [...prev, { role: "assistant", content: assistantText }]);
      }
    } catch (err: any) {
      setMessages((prev) => [...prev, { role: "assistant", content: `Error: ${err.message}` }]);
    } finally {
      setIsLoading(false);
    }
  }, [messages, tripId, onItemAdded]);

  const addSuggestion = useCallback(async (suggestion: Suggestion) => {
    await fetch(`/api/trips/${tripId}/items`, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        name: suggestion.name,
        categorySlug: suggestion.category,
        address: suggestion.address,
        notes: suggestion.notes,
      }),
    });
    onItemAdded();
  }, [tripId, onItemAdded]);

  return { messages, suggestions, isLoading, send, addSuggestion };
}
```

Create `src/app/api/migrate/route.ts`:
```typescript
import { NextRequest, NextResponse } from "next/server";
import { db } from "@/db";
import { trips, tripMembers, days, items, categories } from "@/db/schema";
import { eq, isNull } from "drizzle-orm";
import { getSessionUser } from "@/lib/auth";
import { unauthorized, badRequest } from "@/lib/errors";
import { parseCostString } from "@/lib/cost";
import { parseDateString } from "@/lib/dates";

export async function POST(req: NextRequest) {
  const user = await getSessionUser();
  if (!user) return unauthorized();

  const { trips: legacyTrips } = await req.json();
  if (!legacyTrips || typeof legacyTrips !== "object") {
    return badRequest("Expected { trips: { [id]: { name, items } } }");
  }

  // Load global categories for slug → id mapping
  const globalCats = await db
    .select()
    .from(categories)
    .where(isNull(categories.tripId));
  const catMap = new Map(globalCats.map((c) => [c.slug, c.id]));

  const results: { name: string; success: boolean; error?: string; itemCount?: number }[] = [];

  for (const [, tripData] of Object.entries(legacyTrips) as [string, any][]) {
    try {
      // Create trip
      const [trip] = await db.insert(trips).values({
        name: tripData.name,
      }).returning();

      // Add user as owner
      await db.insert(tripMembers).values({
        tripId: trip.id,
        userId: user.id!,
        role: "owner",
      });

      // Create unscheduled day
      const [unscheduledDay] = await db.insert(days).values({
        tripId: trip.id,
        date: null,
        label: "Unscheduled",
        sortOrder: 9999,
      }).returning();

      // Migrate items
      let count = 0;
      for (const item of tripData.items || []) {
        const categoryId = catMap.get(item.category);
        if (!categoryId) continue; // Skip unknown categories

        const cost = item.cost ? parseCostString(item.cost) : null;
        const startTime = item.time ? parseDateString(item.time) : null;

        await db.insert(items).values({
          tripId: trip.id,
          dayId: unscheduledDay.id,
          categoryId,
          name: item.name,
          address: item.address || null,
          lat: item.lat ? String(item.lat) : null,
          lng: item.lng ? String(item.lng) : null,
          placeId: item.place_id || null,
          startTime: startTime ? new Date(startTime) : null,
          costAmount: cost?.amount ?? null,
          costCurrency: cost?.currency ?? null,
          notes: item.notes || null,
          createdBy: user.id!,
          sortOrder: count,
        });
        count++;
      }

      results.push({ name: tripData.name, success: true, itemCount: count });
    } catch (err: any) {
      results.push({ name: tripData.name, success: false, error: err.message });
    }
  }

  return NextResponse.json({ results });
}
```

```tsx
"use client";

import { useState } from "react";

type MigrationResult = { name: string; success: boolean; error?: string; itemCount?: number };

export default function MigratePage() {
  const [status, setStatus] = useState<"idle" | "loading" | "done">("idle");
  const [results, setResults] = useState<MigrationResult[]>([]);

  const handleMigrate = async () => {
    // Read from localStorage (old site data)
    const raw = localStorage.getItem("tripTerminal");
    if (!raw) {
      alert("No Trip Terminal data found in localStorage.");
      return;
    }

    let data;
    try {
      data = JSON.parse(raw);
      // Handle double-stringified data
      if (typeof data === "string") data = JSON.parse(data);
    } catch {
      alert("Could not parse localStorage data.");
      return;
    }

    if (!data.trips || Object.keys(data.trips).length === 0) {
      alert("No trips found in localStorage data.");
      return;
    }

    setStatus("loading");

    try {
      const res = await fetch("/api/migrate", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ trips: data.trips }),
      });
      const { results } = await res.json();
      setResults(results);
      setStatus("done");
    } catch (err: any) {
      alert(`Migration failed: ${err.message}`);
      setStatus("idle");
    }
  };

  return (
    <div className="migrate-page">
      <h1>Migrate Your Trips</h1>
      <p>Transfer your trips from the old Trip Terminal to the new version.</p>

      {status === "idle" && (
        <button onClick={handleMigrate} className="btn-primary">
          Start Migration
        </button>
      )}

      {status === "loading" && <p>Migrating your trips...</p>}

      {status === "done" && (
        <div className="migration-results">
          <h2>Migration Complete</h2>
          {results.map((r, i) => (
            <div key={i} className={`migration-result ${r.success ? "success" : "error"}`}>
              <span>{r.success ? "✅" : "❌"} {r.name}</span>
              {r.success && <span className="dim"> — {r.itemCount} items</span>}
              {r.error && <span className="error"> — {r.error}</span>}
            </div>
          ))}
          <a href="/dashboard" className="btn-primary" style={{ marginTop: "1rem", display: "inline-block" }}>
            Go to Dashboard →
          </a>
        </div>
      )}
    </div>
  );
}
```

```tsx
"use client";

import { useEffect, useCallback, type ReactNode } from "react";

type Props = {
  isOpen: boolean;
  onClose: () => void;
  children: ReactNode;
  title?: string;
};

export function Modal({ isOpen, onClose, children, title }: Props) {
  const handleEsc = useCallback((e: KeyboardEvent) => {
    if (e.key === "Escape") onClose();
  }, [onClose]);

  useEffect(() => {
    if (isOpen) document.addEventListener("keydown", handleEsc);
    return () => document.removeEventListener("keydown", handleEsc);
  }, [isOpen, handleEsc]);

  if (!isOpen) return null;

  return (
    <div className="modal-overlay" onClick={onClose}>
      <div className="modal-content" onClick={(e) => e.stopPropagation()}>
        {title && (
          <div className="modal-header">
            <h2>{title}</h2>
            <button className="modal-close" onClick={onClose}>✕</button>
          </div>
        )}
        <div className="modal-body">{children}</div>
      </div>
    </div>
  );
}
```
